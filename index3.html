import tkinter as tk class ButtonActivity(tk.Tk): def __init__(self): super().__init__() self.title("Counter Button") # Initialize counter self.counter = 0 # Create and pack label to show counter value self.label_counter = tk.Label(self, name="label_counter", text="0") # Default text as "0" self.label_counter.pack() # Create and pack button to increment counter self.button_count = tk.Button(self, name="button_count", text="Increment", command=self.increment_counter) self.button_count.pack() def increment_counter(self): self.counter += 1 self.label_counter.config(text=str(self.counter)) # Update label with string representation of counter if __name__ == "__main__": app = ButtonActivity() app.mainloop()
